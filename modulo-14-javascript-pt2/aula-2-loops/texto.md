## For
var alunos = [
    [8,1,5,6],
    [8,6,1,5],
    [6,4,8,2],
]
var nota = 0;
for(let i = 0; i < alunos.length; i++){
    for(c = 0; c < aluno[i];c++){
        nota += aluno[c];
    }
    media = nota / aluno[c].length

    if(media >= 7){
        console.log("Aprovado")
    }else{
        console.log("Reprovado")
    }
}
